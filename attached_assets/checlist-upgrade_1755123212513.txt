    <script>
        // Função para mostrar/ocultar campo de "Outra" empreiteira
        function toggleField(selectId, triggerValue, fieldId) {
            const selectElement = document.getElementById(selectId);
            const fieldElement = document.getElementById(fieldId);
            if (selectElement.value === triggerValue) {
                fieldElement.classList.remove('hidden');
            } else {
                fieldElement.classList.add('hidden');
            }
        }

        // Função para mostrar/ocultar campos de evidência (Foto ou Observação)
        function toggleEvidence(radioElement, photoDivId, obsDivId) {
            const photoDiv = document.getElementById(photoDivId);
            const obsDiv = document.getElementById(obsDivId);
            
            if (radioElement.value === 'sim') {
                photoDiv.classList.remove('hidden');
                obsDiv.classList.add('hidden');
            } else {
                photoDiv.classList.add('hidden');
                obsDiv.classList.remove('hidden');
            }A
        }
    </script>
